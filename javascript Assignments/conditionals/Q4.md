// Q4. Generate numbers between any 2 given numbers.
// Ex
// 8 onst num1 = 10
// 8 onst num2 = 25;
// O9tp9t: 11, 12, 13, ...., 25


function generateNumbers(start, end) {
    const numbers = [];
    for (let i = start + 1; i <= end; i++) {
      numbers.push(i);
    }
    return numbers;
  }
  
  // Example usage:
  const num1 = 10;
  const num2 = 25;
  const result = generateNumbers(num1, num2);
  console.log(result); // Output: [11, 12, 13, ..., 25]
